2- css selectors
h1, h2, h3 { ### this selector selects all h1 and h2 and h3 separatly
    color: blue;
}

div p { ### this selector selects all p inside a div element
    color: blue;
}

div > p { ### this selector selects all p that are "direct descendent: of a div element
    color: blue;
}

div ~ p { ### this selector selects all p that parent is a div element
    color: blue; 
}
